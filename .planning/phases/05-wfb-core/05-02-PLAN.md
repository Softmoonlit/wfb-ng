# Plan 05-02: TX 发射线程整合

## Objective

将 tx.cpp 中的发射逻辑整合到 wfb_core 中，实现 tx_main_loop 负责零轮询唤醒、门控控制、数据包发射。

## Background

当前 tx.cpp 是独立的可执行文件，需要将其核心逻辑抽取为线程函数，在 wfb_core 进程内运行。发射线程需要响应 Token 授权，在授权时间窗口内发射数据包。

### 2. 创建 tx_worker.cpp

```cpp
#include "tx_worker.h"
#include "packet_queue.h"
#include "watermark.h"
#include "radiotap_template.h"
#include "guard_interval.h"
#include <cstring>
#include <iostream>

namespace {

// 从队列中批量取包
size_t dequeue_packets(PacketQueue& queue,
                       std::vector<PacketQueue::Packet>& packets,
                       size_t max_packets) {
    size_t count = 0;
    PacketQueue::Packet pkt;
    while (count < max_packets && queue.try_pop(pkt)) {
        packets.push_back(std::move(pkt));
        count++;
    }
    return count;
}

// 构建 IEEE 802.11 帧
std::vector<uint8_t> build_frame(const PacketQueue::Packet& pkt,
                                  const RadiotapTemplate& rtap_template) {
    std::vector<uint8_t> frame;
    // Radiotap 头部
    const uint8_t* rtap_data = rtap_template.get_template();
    uint16_t rtap_len = rtap_template.get_length();
    frame.insert(frame.end(), rtap_data, rtap_data + rtap_len);

    // IEEE 802.11 数据帧头部（简化版）
    // Frame Control: Data frame, To DS
    frame.push_back(0x08);  // Data frame
    frame.push_back(0x01);  // To DS
    // Duration
    frame.push_back(0x00);
    frame.push_back(0x00);
    // Address 1-4 (BSSID, SA, DA, BSSID)
    // 使用广播地址简化
    for (int i = 0; i < 6; i++) frame.push_back(0xFF);
    for (int i = 0; i < 6; i++) frame.push_back(0x00);
    for (int i = 0; i < 6; i++) frame.push_back(0xFF);
    for (int i = 0; i < 6; i++) frame.push_back(0x00);
    // Sequence Control
    frame.push_back(0x00);
    frame.push_back(0x00);

    // 数据载荷
    frame.insert(frame.end(), pkt.data, pkt.data + pkt.len);

    return frame;
}

}  // namespace

void tx_main_loop(ThreadSharedState* shared_state,
                  pcap_t* pcap_handle,
                  std::atomic<bool>* shutdown,
                  int mcs,
                  int bandwidth) {
    // 初始化 Radiotap 模板
    RadiotapTemplate rtap_template;
    rtap_template.init_for_data(mcs, bandwidth);

    // 批量发射参数
    const size_t kBatchSize = 32;
    std::vector<PacketQueue::Packet> batch;
    batch.reserve(kBatchSize);

    while (!shutdown->load()) {
        std::unique_lock<std::mutex> lock(shared_state->mtx);

        // 零轮询唤醒：等待 can_send 或 shutdown
        shared_state->cv_send.wait(lock, [&]() {
            return shared_state->can_send || shutdown->load();
        });

        if (shutdown->load()) break;

        // 门控检查
        uint64_t now_ms = get_monotonic_ms();
        if (!shared_state->can_send ||
            now_ms > shared_state->token_expire_time_ms) {
            shared_state->can_send = false;
            continue;  // 继续等待下一个 Token
        }

        // 计算剩余授权时间
        uint64_t remaining_ms = shared_state->token_expire_time_ms - now_ms;

        // 解锁，准备发射
        lock.unlock();

        // 批量取包
        batch.clear();
        size_t count = dequeue_packets(shared_state->packet_queue, batch, kBatchSize);

        // 发射数据包
        for (size_t i = 0; i < count; i++) {
            // 检查 Token 是否过期
            if (get_monotonic_ms() > shared_state->token_expire_time_ms) {
                // Token 过期，停止发射
                // 将未发射的包放回队列（简化：直接丢弃）
                break;
            }

            auto frame = build_frame(batch[i], rtap_template);
            int ret = pcap_inject(pcap_handle, frame.data(), frame.size());
            if (ret < 0) {
                std::cerr << "pcap_inject 失败: "
                          << pcap_geterr(pcap_handle) << std::endl;
            }
        }

        // 应用保护间隔（节点切换时使用）
        apply_guard_interval(1);  // 1ms 微间隔
    }
}
```
