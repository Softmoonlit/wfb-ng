```cpp
#include <gtest/gtest.h>
#include <gmock/gmock.h>
#include <thread>
#include <chrono>
#include <atomic>

#include "threads.h"
#include "packet_queue.h"
#include "watermark.h"
#include "mac_token.h"

// 测试线程启动和退出
TEST(WfbCoreTest, ThreadLifecycle) {
    ThreadSharedState state;
    std::atomic<bool> shutdown{false};

    // 启动模拟线程
    std::thread control_thread([&]() {
        while (!shutdown.load()) {
            std::unique_lock<std::mutex> lock(state.mtx);
            state.can_send = true;
            state.token_expire_time_ms = 0;
            state.cv_send.notify_one();
            lock.unlock();
            std::this_thread::sleep_for(std::chrono::milliseconds(10));
        }
    });

    // 运行 100ms
    std::this_thread::sleep_for(std::chrono::milliseconds(100));

    // 触发退出
    shutdown.store(true);
    control_thread.join();

    EXPECT_TRUE(true);  // 线程正常退出
}

// 测试 Token 授权和过期
TEST(WfbCoreTest, TokenGating) {
    ThreadSharedState state;

    // 模拟 Token 授权
    {
        std::lock_guard<std::mutex> lock(state.mtx);
        state.can_send = true;
        state.token_expire_time_ms = get_monotonic_ms() + 50;  // 50ms
    }

    // 立即检查：应允许发射
    {
        std::lock_guard<std::mutex> lock(state.mtx);
        EXPECT_TRUE(state.can_send);
        EXPECT_GT(state.token_expire_time_ms, get_monotonic_ms());
    }

    // 等待 Token 过期
    std::this_thread::sleep_for(std::chrono::milliseconds(60));

    // 检查：Token 应已过期
    {
        std::lock_guard<std::mutex> lock(state.mtx);
        EXPECT_GT(get_monotonic_ms(), state.token_expire_time_ms);
    }
}

// 测试环形队列反压
TEST(WfbCoreTest, QueueBackpressure) {
    PacketQueue queue;
    queue.set_capacity(100);

    // 填满队列
    PacketQueue::Packet pkt;
    memset(pkt.data, 0xAA, sizeof(pkt.data));
    pkt.len = 100;

    int push_count = 0;
    while (queue.push(pkt)) {
        push_count++;
        if (push_count > 150) break;  // 安全限制
    }

    // 队列满时应拒绝 push
    EXPECT_GE(push_count, 100);
    EXPECT_LT(push_count, 150);

    // 检查队列大小
    EXPECT_EQ(queue.size(), static_cast<size_t>(queue.capacity()));
}

// 测试动态水位线计算
TEST(WfbCoreTest, DynamicWatermark) {
    // MCS 0, 20MHz, 50ms 窗口
    uint32_t limit1 = calculate_dynamic_limit(50, 6000000, 1450);
    EXPECT_GT(limit1, 0u);
    EXPECT_LT(limit1, 1000u);  // 应在合理范围内

    // MCS 6, 20MHz, 50ms 窗口（更高速率）
    uint32_t limit2 = calculate_dynamic_limit(50, 54000000, 1450);
    EXPECT_GT(limit2, limit1);  // 应比低速时更大

    // 验证公式：limit = ceil((R_phy × duration × 1.5) / (8 × MTU × 1000))
    double expected = std::ceil((6000000.0 * 50 * 1.5) / (8 * 1450 * 1000));
    EXPECT_NEAR(limit1, expected, 1.0);
}

// 测试 TokenFrame 序列化和反序列化
TEST(WfbCoreTest, TokenFrameSerialization) {
    TokenFrame original;
    original.magic = 0x02;
    original.target_node = 5;
    original.time_quota_ms = 300;
    original.seq_number = 12345;

    // 序列化
    uint8_t buffer[TokenFrame::kSerializedSize];
    ASSERT_TRUE(original.serialize(buffer, sizeof(buffer)));

    // 反序列化
    TokenFrame decoded;
    ASSERT_TRUE(decoded.deserialize(buffer, sizeof(buffer)));

    // 验证字段
    EXPECT_EQ(decoded.magic, original.magic);
    EXPECT_EQ(decoded.target_node, original.target_node);
    EXPECT_EQ(decoded.time_quota_ms, original.time_quota_ms);
    EXPECT_EQ(decoded.seq_number, original.seq_number);
}

// 测试 condition_variable 唤醒
TEST(WfbCoreTest, ConditionVariableWakeup) {
    ThreadSharedState state;
    std::atomic<bool> woke_up{false};

    // 等待线程
    std::thread waiter([&]() {
        std::unique_lock<std::mutex> lock(state.mtx);
        state.cv_send.wait(lock, [&]() {
            return state.can_send;
        });
        woke_up.store(true);
    });

    // 等待 10ms
    std::this_thread::sleep_for(std::chrono::milliseconds(10));

    // 唤醒线程
    {
        std::lock_guard<std::mutex> lock(state.mtx);
        state.can_send = true;
        state.cv_send.notify_one();
    }

    // 等待线程响应
    waiter.join();

    EXPECT_TRUE(woke_up.load());
}

int main(int argc, char** argv) {
    testing::InitGoogleTest(&argc, argv);
    return RUN_ALL_TESTS();
}
```

### 2. 创建 integration_test_v2.sh

```bash
#!/bin/bash

# Phase 5 集成测试脚本
# 测试单进程架构的端到端功能

set -e

echo "=== Phase 5 集成测试 ==="

# 1. 编译测试
echo "[1/6] 编译测试..."
make clean
make test_wfb_core

# 2. 运行单元测试
echo "[2/6] 运行单元测试..."
./tests/test_wfb_core

# 3. 启动模拟服务端（后台）
echo "[3/6] 启动模拟服务端..."
sudo ./build/wfb_core --mode server -i wlan0 -c 6 -m 0 --tun wfb0 &
SERVER_PID=$!
sleep 2

# 4. 启动模拟客户端（后台）
echo "[4/6] 启动模拟客户端..."
sudo ./build/wfb_core --mode client -i wlan1 -c 6 -m 0 --tun wfb1 --node-id 1 &
CLIENT_PID=$!
sleep 2

# 5. 验证进程运行状态
echo "[5/6] 验证进程状态..."
if ! kill -0 $SERVER_PID 2>/dev/null; then
    echo "✗ 服务端进程未运行"
    exit 1
fi

if ! kill -0 $CLIENT_PID 2>/dev/null; then
    echo "✗ 客户端进程未运行"
    exit 1
fi

echo "✓ 服务端和客户端进程正常运行"

# 6. 清理
echo "[6/6] 清理测试环境..."
sudo kill -TERM $SERVER_PID $CLIENT_PID
wait $SERVER_PID $CLIENT_PID 2>/dev/null

echo "=== 集成测试完成 ==="
```
