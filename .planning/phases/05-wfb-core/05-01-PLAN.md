### 2. 创建 rx_demux.cpp

```cpp
#include "rx_demux.h"
#include "mac_token.h"
#include <cstring>
#include <iostream>

void rx_demux_main_loop(ThreadSharedState* shared_state,
                        pcap_t* pcap_handle,
                        std::atomic<bool>* shutdown) {
    const int kPcapTimeoutMs = 10;  // 10ms 超时，便于检查 shutdown

    while (!shutdown->load()) {
        struct pcap_pkthdr* header;
        const u_char* data;

        int ret = pcap_next_ex(pcap_handle, &header, &data);
        if (ret <= 0) {
            if (ret == -1) {
                std::cerr << "pcap 错误: " << pcap_geterr(pcap_handle) << std::endl;
            }
            continue;  // 超时或无数据
        }

        // 跳过 Radiotap 头部
        if (header->cap_len < sizeof(ieee80211_radiotap_header)) {
            continue;
        }

        auto* rtap = reinterpret_cast<const ieee80211_radiotap_header*>(data);
        uint16_t rtap_len = rtap->it_len;

        if (header->cap_len < rtap_len + sizeof(TokenFrame)) {
            continue;  // 数据太短
        }

        // 解析 TokenFrame
        const uint8_t* token_data = data + rtap_len;
        TokenFrame token;
        if (!token.deserialize(token_data, header->cap_len - rtap_len)) {
            continue;  // 解析失败
        }

        // 验证魔数
        if (token.magic != 0x02) {
            continue;
        }

        // 更新共享状态（Server 模式）
        {
            std::lock_guard<std::mutex> lock(shared_state->mtx);
            shared_state->can_send = true;
            shared_state->token_expire_time_ms =
                get_monotonic_ms() + token.time_quota_ms;
            shared_state->cv_send.notify_one();
        }

        // TODO: 处理数据帧（写入 TUN）
    }
}
```
