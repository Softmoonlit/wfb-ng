# Plan 05-03: TUN 读取线程整合

## Objective

将 wfb_tun.c 中的 TUN 设备处理逻辑整合到 wfb_core 中，实现 tun_reader_thread 负责从 TUN 设备读取数据、FEC 编码、写入环形队列。

## Background

当前 wfb_tun.c 是独立的 C 程序，需要将其核心逻辑抽取为 C++ 线程函数，在 wfb_core 进程内运行。TUN 读取线程需要制造反压，当队列满时停止读取。

### 2. 创建 tun_reader.cpp

```cpp
#include "tun_reader.h"
#include "watermark.h"
#include "zfex.h"

#include <fcntl.h>
#include <linux/if.h>
#include <linux/if_tun.h>
#include <sys/ioctl.h>
#include <unistd.h>

#include <cstring>
#include <iostream>

namespace {

// TUN 设备创建和配置
int create_tun_device(const std::string& name) {
    int fd = open("/dev/net/tun", O_RDWR);
    if (fd < 0) {
        std::cerr << "无法打开 /dev/net/tun: " << strerror(errno) << std::endl;
        return -1;
    }

    struct ifreq ifr;
    memset(&ifr, 0, sizeof(ifr));
    ifr.ifr_flags = IFF_TUN | IFF_NO_PI;
    strncpy(ifr.ifr_name, name.c_str(), IFNAMSIZ - 1);

    if (ioctl(fd, TUNSETIFF, &ifr) < 0) {
        std::cerr << "TUNSETIFF 失败: " << strerror(errno) << std::endl;
        close(fd);
        return -1;
    }

    return fd;
}

// 设置 TUN 队列长度（极小水池反压）
bool set_tun_txqueuelen(const std::string& name, int qlen) {
    // 通过系统调用 ip 命令设置
    std::string cmd = "ip link set dev " + name + " txqueuelen " + std::to_string(qlen);
    int ret = system(cmd.c_str());
    return (ret == 0);
}

// FEC 编码参数
constexpr int kFEC_N = 12;  // 总块数
constexpr int kFEC_K = 8;   // 数据块数

}  // namespace

void tun_reader_main_loop(ThreadSharedState* shared_state,
                          const std::string& tun_name,
                          std::atomic<bool>* shutdown,
                          int mcs,
                          int bandwidth) {
    // 创建 TUN 设备
    int tun_fd = create_tun_device(tun_name);
    if (tun_fd < 0) {
        std::cerr << "TUN 设备创建失败" << std::endl;
        return;
    }

    // 设置极小队列长度（反压关键）
    if (!set_tun_txqueuelen(tun_name, 100)) {
        std::cerr << "警告: 设置 txqueuelen 失败" << std::endl;
    }

    // 初始化 FEC 编码器
    zfex_status status;
    zfex_code* fec_code = zfex_code_create(&status, kFEC_N, kFEC_K);
    if (!fec_code) {
        std::cerr << "FEC 编码器创建失败" << std::endl;
        close(tun_fd);
        return;
    }

    // 计算动态水位线
    // 假设 50ms 授权窗口，1.5 倍超发系数
    const uint32_t dynamic_limit = calculate_dynamic_limit(
        50, get_phy_rate_bps(mcs, bandwidth), 1450);

    // FEC 块缓冲区
    std::vector<std::vector<uint8_t>> fec_blocks(kFEC_N);
    for (auto& block : fec_blocks) {
        block.resize(1450);
    }
    std::vector<const uint8_t*> fec_inputs(kFEC_K);
    std::vector<uint8_t*> fec_outputs(kFEC_N - kFEC_K);

    size_t block_count = 0;

    while (!shutdown->load()) {
        // 检查队列是否达到水位线
        if (shared_state->packet_queue.size() >= dynamic_limit) {
            // 队列满，制造反压：休眠让出 CPU
            usleep(1000);  // 1ms 退避
            continue;
        }

        // 读取 TUN 数据
        ssize_t nread = read(tun_fd, fec_blocks[block_count].data(), 1450);
        if (nread < 0) {
            if (errno == EAGAIN || errno == EWOULDBLOCK) {
                continue;
            }
            std::cerr << "TUN 读取错误: " << strerror(errno) << std::endl;
            break;
        }

        // 记录实际长度
        fec_blocks[block_count].resize(static_cast<size_t>(nread));
        fec_inputs[block_count] = fec_blocks[block_count].data();
        block_count++;

        // 当收集到 K 个块时进行 FEC 编码
        if (block_count >= static_cast<size_t>(kFEC_K)) {
            // 准备输出缓冲区
            for (size_t i = 0; i < static_cast<size_t>(kFEC_N - kFEC_K); i++) {
                fec_outputs[i] = fec_blocks[kFEC_K + i].data();
            }

            // FEC 编码
            zfex_code_encode(fec_code,
                             const_cast<const uint8_t**>(fec_inputs.data()),
                             fec_outputs.data());

            // 将所有块（数据块 + 冗余块）入队
            for (size_t i = 0; i < static_cast<size_t>(kFEC_N); i++) {
                PacketQueue::Packet pkt;
                pkt.len = (i < static_cast<size_t>(kFEC_K))
                              ? fec_blocks[i].size()
                              : 1450;  // 冗余块固定大小
                memcpy(pkt.data, fec_blocks[i].data(), pkt.len);

                if (!shared_state->packet_queue.push(pkt)) {
                    // 队列满，丢弃
                    break;
                }
            }

            // 重置块计数
            block_count = 0;

            // 通知发射线程有新数据
            {
                std::lock_guard<std::mutex> lock(shared_state->mtx);
                shared_state->cv_send.notify_one();
            }
        }
    }

    // 清理
    zfex_code_destroy(fec_code);
    close(tun_fd);
}
```
