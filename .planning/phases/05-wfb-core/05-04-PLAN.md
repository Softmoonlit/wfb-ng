### 2. 创建 scheduler_worker.cpp

```cpp
#include "scheduler_worker.h"
#include "mac_token.h"
#include "token_emitter.h"
#include "breathing_cycle.h"

#include <iostream>

void scheduler_main_loop(ThreadSharedState* shared_state,
                         ServerScheduler* scheduler,
                         pcap_t* pcap_handle,
                         std::atomic<bool>* shutdown) {
    // 初始化呼吸周期（下行广播）
    BreathingCycle breathing_cycle;

    // Token 发射器
    TokenEmitter token_emitter(pcap_handle);

    // 主调度循环
    while (!shutdown->load()) {
        // 1. 获取下一个要调度的节点
        uint8_t next_node = scheduler->get_next_scheduled_node();

        if (next_node == 0) {
            // 全员深睡，执行空闲巡逻
            scheduler->idle_patrol();
            usleep(100000);  // 100ms 间隔
            continue;
        }

        // 2. 计算授权时间窗口
        uint32_t time_quota_ms = scheduler->calculate_time_quota(next_node);

        // 3. 构建 TokenFrame
        TokenFrame token;
        token.magic = 0x02;
        token.target_node = next_node;
        token.time_quota_ms = time_quota_ms;
        token.seq_number = token_emitter.get_next_seq();

        // 4. 三连发 Token（击穿机制）
        if (!token_emitter.send_token_triple(token)) {
            std::cerr << "Token 发射失败" << std::endl;
            continue;
        }

        // 5. 应用保护间隔（节点切换）
        apply_guard_interval(5);  // 5ms

        // 6. 更新共享状态（供发射线程使用）
        {
            std::lock_guard<std::mutex> lock(shared_state->mtx);
            shared_state->can_send = true;
            shared_state->token_expire_time_ms =
                get_monotonic_ms() + time_quota_ms;
            shared_state->cv_send.notify_one();
        }

        // 7. 等待授权窗口过期
        usleep(time_quota_ms * 1000);

        // 8. 更新节点状态
        scheduler->update_node_state(next_node);

        // 9. 呼吸周期管理（仅服务端）
        breathing_cycle.advance_phase();
        if (breathing_cycle.is_inhale_phase()) {
            // 吸气阶段：轮询所有节点的 NACK
            for (uint8_t node_id = 1; node_id <= scheduler->get_node_count(); node_id++) {
                scheduler->schedule_node_for_nack(node_id, 25);  // 25ms NACK 窗口
            }
        }
    }
}
```

### 3. 更新 wfb_core.cpp

在 run_server() 中添加调度器线程启动：

```cpp
#include "scheduler_worker.h"
#include "server_scheduler.h"

// 在 run_server() 中
ServerScheduler scheduler(config.mcs, config.bandwidth);

// 初始化节点列表（从配置文件或命令行）
// 假设有 10 个节点，ID 从 1 到 10
for (uint8_t node_id = 1; node_id <= 10; node_id++) {
    scheduler.register_node(node_id);
}

// 启动调度器线程
auto scheduler_thread = start_thread("scheduler", 95, [&]() {
    scheduler_main_loop(&shared_state, &scheduler, pcap_handle, &g_shutdown);
});
```

### 4. 调度器接口确认

确保 ServerScheduler 提供以下接口：
- `void register_node(uint8_t node_id)` — 注册节点
- `uint8_t get_next_scheduled_node()` — 获取下一个调度节点（0 表示全员深睡）
- `uint32_t calculate_time_quota(uint8_t node_id)` — 计算授权时间窗口
- `void idle_patrol()` — 空闲巡逻（探询 SQ 队首）
- `void update_node_state(uint8_t node_id)` — 更新节点状态机
- `size_t get_node_count()` — 获取节点总数
- `void schedule_node_for_nack(uint8_t node_id, uint32_t time_ms)` — 安排 NACK 回传窗口
