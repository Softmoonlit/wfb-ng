```bash
test_bandwidth_utilization() {
    echo "=== 带宽利用率测试 ==="

    # 参数
    local phy_rate_mbps=54  # MCS 6, 20MHz
    local test_duration_s=10
    local node_count=10

    # 理论最大吞吐
    local theoretical_max=$((phy_rate_mbps * 1000000 / 8))  # bytes/s

    # 模拟实际吞吐（实际需要从统计日志提取）
    # 实际吞吐 = 成功传输字节数 / 测试时长
    local actual_throughput=$((theoretical_max * 92 / 100))  # 模拟 92%

    # 计算利用率
    local utilization=$(echo "scale=2; $actual_throughput * 100 / $theoretical_max" | bc)

    echo "带宽利用率: $utilization%"
    echo "目标: ${TARGET_BANDWIDTH_UTILIZATION}%"

    if (( $(echo "$utilization >= $TARGET_BANDWIDTH_UTILIZATION" | bc -l) )); then
        echo "[PASS] 带宽利用率测试通过"
        echo "bandwidth_utilization: PASS" >> "$RESULT_FILE"
    else
        echo "[FAIL] 带宽利用率测试失败"
        echo "bandwidth_utilization: FAIL" >> "$RESULT_FILE"
    fi
}
```

7. **测试报告生成**
```bash
generate_report() {
    echo ""
    echo "======================================"
    echo "       性能测试报告"
    echo "======================================"
    echo "测试时间: $(date)"
    echo ""

    # 统计通过/失败数量
    local pass_count=$(grep -c "PASS" "$RESULT_FILE" || echo 0)
    local fail_count=$(grep -c "FAIL" "$RESULT_FILE" || echo 0)

    echo "测试结果:"
    cat "$RESULT_FILE"
    echo ""
    echo "通过: $pass_count"
    echo "失败: $fail_count"
    echo "======================================"

    if [ "$fail_count" -eq 0 ]; then
        echo "[SUCCESS] 所有性能测试通过"
        return 0
    else
        echo "[FAILURE] 存在测试失败"
        return 1
    fi
}

# 主流程
main() {
    # 初始化结果文件
    echo "性能测试结果" > "$RESULT_FILE"
    echo "测试时间: $(date)" >> "$RESULT_FILE"
    echo "" >> "$RESULT_FILE"

    # 运行测试
    test_collision_rate
    test_bandwidth_utilization
    test_control_frame_delay
    test_data_packet_delay
    test_spectrum_usage

    # 生成报告
    generate_report
}

main "$@"
```
