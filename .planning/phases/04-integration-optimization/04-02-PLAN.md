```bash
check_root() {
    if [[ $EUID -ne 0 ]]; then
        echo "错误: 必须以 Root 权限运行此脚本" >&2
        echo "提示: 使用 sudo $0" >&2
        exit 1
    fi
}
```

4. **网卡 Monitor 模式检查**
```bash
check_monitor_mode() {
    local iface="$1"
    if ! iwconfig "$iface" 2>&1 | grep -q "Mode:Monitor"; then
        echo "错误: 网卡 $iface 未处于 Monitor 模式" >&2
        echo "提示: 使用 'iwconfig $iface mode Monitor' 设置" >&2
        exit 1
    fi
    echo "检查通过: 网卡 $iface 处于 Monitor 模式"
}
```

5. **TUN 设备设置**
```bash
setup_tun() {
    local tun_name="$1"
    local txqueuelen=100
    
    # 创建 TUN 设备
    ip tuntap add dev "$tun_name" mode tun
    ip link set "$tun_name" up
    ip link set "$tun_name" txqueuelen $txqueuelen
    
    echo "TUN 设备 $tun_name 已创建，txqueuelen=$txqueuelen"
}
```

6. **日志目录创建**
```bash
setup_logging() {
    local log_dir=$(dirname "$LOG_FILE")
    mkdir -p "$log_dir"
    echo "日志目录已创建: $log_dir"
}
```

7. **主程序启动**
```bash
start_server() {
    echo "启动 WFB-NG 服务器..."
    echo "配置: MCS=$MCS_INDEX, 频宽=${BANDWIDTH}MHz, 节点数=$NUM_NODES"
    
    # 启动主程序（需要根据实际可执行文件调整）
    # ./wfb_core --mcs $MCS_INDEX --bandwidth $BANDWIDTH --nodes $NUM_NODES 2>&1 | tee "$LOG_FILE"
    echo "提示: 请根据实际可执行文件路径调整启动命令"
}

# 主流程
main() {
    check_root
    check_monitor_mode "$WLAN_IFACE"
    setup_tun "$TUN_NAME"
    setup_logging
    start_server
}

main "$@"
```
