3. **多节点并发测试**
```cpp
TEST_F(IntegrationTest, MultiNodeConcurrencyTest) {
    // 测试目标：验证多节点并发调度正确性
    const size_t NUM_NODES = 10;

    // 1. 初始化 AQ/SQ 管理器
    AqSqManager manager;
    for (size_t i = 1; i <= NUM_NODES; ++i) {
        manager.add_node_to_aq(static_cast<uint8_t>(i));
    }

    // 2. 初始化调度器
    ServerScheduler scheduler;
    scheduler.set_num_nodes(NUM_NODES);
    scheduler.set_aq_sq_manager(&manager);

    for (size_t i = 1; i <= NUM_NODES; ++i) {
        scheduler.init_node(static_cast<uint8_t>(i));
    }

    // 3. 模拟多轮调度
    std::vector<uint8_t> served_nodes;
    for (int round = 0; round < 20; ++round) {
        auto next = scheduler.get_next_node_to_serve();
        if (next.first != 0xFF) {
            served_nodes.push_back(next.first);
        }
    }

    // 4. 验证所有节点都被服务
    EXPECT_GE(served_nodes.size(), NUM_NODES);
}
```

4. **40MB 模型传输模拟**
```cpp
TEST_F(IntegrationTest, ModelTransferSimulation) {
    // 测试目标：模拟 40MB 模型传输场景
    const size_t MODEL_SIZE = 40 * 1024 * 1024;  // 40MB
    const size_t MTU = 1450;
    const size_t TOTAL_PACKETS = MODEL_SIZE / MTU;

    // 1. 初始化队列
    ThreadSafeQueue queue;

    // 2. 模拟数据包入队
    auto start_time = std::chrono::steady_clock::now();
    for (size_t i = 0; i < TOTAL_PACKETS; ++i) {
        queue.push(static_cast<uint8_t>(i & 0xFF));
    }
    auto end_time = std::chrono::steady_clock::now();

    // 3. 验证入队时间（应小于 1 秒）
    auto duration_ms = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
    EXPECT_LT(duration_ms.count(), 1000);

    // 4. 验证队列大小
    EXPECT_EQ(queue.size(), TOTAL_PACKETS);
}
```

TEST_F(IntegrationTest, TokenFrameIntegration) {
    // 测试目标：验证 TokenFrame 在集成场景下的正确性
    TokenFrame frame;
    frame.magic = 0x02;
    frame.target_node = 5;
    frame.duration_ms = 100;
    frame.seq_num = 42;

    // 1. 序列化
    std::vector<uint8_t> buffer;
    serialize_token_frame(frame, buffer);

    // 2. 反序列化
    TokenFrame decoded;
    deserialize_token_frame(buffer, decoded);

    // 3. 验证一致性
    EXPECT_EQ(decoded.magic, frame.magic);
    EXPECT_EQ(decoded.target_node, frame.target_node);
    EXPECT_EQ(decoded.duration_ms, frame.duration_ms);
    EXPECT_EQ(decoded.seq_num, frame.seq_num);
}
