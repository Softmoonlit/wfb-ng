TEST_F(IntegrationTest, EndToEndTransferValidation) {
    // 测试目标：验证端到端传输流程正确

    // 1. 初始化完整系统组件
    ServerScheduler scheduler;
    scheduler.set_num_nodes(5);

    AqSqManager aq_manager;
    for (size_t i = 1; i <= 5; ++i) {
        aq_manager.add_node_to_aq(static_cast<uint8_t>(i));
        scheduler.init_node(static_cast<uint8_t>(i));
    }
    scheduler.set_aq_sq_manager(&aq_manager);

    // 2. 初始化数据队列
    ThreadSafeQueue data_queue;

    // 3. 模拟数据传输流程
    // - TUN 读取线程读取数据包
    // - 数据包进入队列
    // - 调度器分配 Token
    // - 发射线程发送数据

    for (int i = 0; i < 100; ++i) {
        data_queue.push(static_cast<uint8_t>(i));
    }

    // 4. 验证数据传输
    EXPECT_EQ(data_queue.size(), 100) << "数据包入队成功";

    // 5. 验证调度器工作
    auto next_node = scheduler.get_next_node_to_serve();
    EXPECT_NE(next_node.first, 0xFF) << "调度器返回有效节点";
}

## 零感知验证

### UFTP 无需修改源码

UFTP 继续使用标准 UDP/IP 协议，无需修改源码：
- 组播地址：224.1.1.1
- 端口：1042
- 协议：UDP/IP

### 底层调度透明性

底层 Token Passing 调度对 UFTP 完全透明：
1. **TUN 设备路由**：特定 IP 段路由到 TUN 设备，不影响其他应用
2. **自动调度**：服务器自动分配 Token，客户端无需感知
3. **快速反压**：TUN 队列满时自动丢包，UFTP 收到 ENOBUFS 信号

### 验证方法

1. **网络抓包分析**
```bash
# 服务端抓包
tcpdump -i wlan0 -w server.pcap

# 客户端抓包
tcpdump -i wlan0 -w client.pcap

# 分析抓包文件
wireshark server.pcap
```

2. **应用层日志分析**
```bash
# UFTP 日志
uftp -R 810 -g 224.1.1.1 -p 1042 model.bin 2>&1 | tee uftp.log

# 底层日志
tail -f /var/log/wfb-ng/server.log
```
